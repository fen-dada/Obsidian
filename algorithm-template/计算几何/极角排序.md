struct Point {
    long double x, y;
    long double angle;
    int id;
};

void polar_sort(vector<Point>& pts) {
    for (auto& p : pts) {
        p.angle = atan2l(p.y, p.x);
        // 如果是负数（三四象限），直接加 2π，把范围强行拉到 [0, 2π)
        if (p.angle < 0) {
            p.angle += 2.0 * acosl(-1.0);
        }
    }
    
    sort(pts.begin(), pts.end(), [](const Point& a, const Point& b) {
        if (abs(a.angle - b.angle) > 1e-12) {
            return a.angle < b.angle;
        }
        return (a.x * a.x + a.y * a.y) < (b.x * b.x + b.y * b.y);
    });
}
