#include <bits/stdc++.h>
using namespace std;
using ll = long long;
#define int ll
int a[100];
int basis[55];
bool insert(int x){
    for(int i=49;i>=0;--i){
        if(x>>i&1){
            if(basis[i]==0){
                basis[i] = x;
                return true;
            }else{
                x^=basis[i];
            }
        }
    }
    return false;
}
void solve(){
    int n;
    cin>>n;
    for(int i=1;i<=n;++i){
        cin>>a[i];
        insert(a[i]);
    } 
    int ans = 0;
    for(int i=50-1;i>=0;--i){
        ans = max(ans,ans^basis[i]);
    }
    cout<<ans;
}
signed main(){
    ios::sync_with_stdio(0);
    cin.tie(0),cout.tie(0);
    int T=1;
    while(T--){
        solve();
    }
}
