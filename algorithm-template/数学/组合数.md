#define int long long
const int mod = 1e9 + 7;
const int N = 1e6+10;
int f[N], g[N];
const int M = 1e9+7;
int qpow(int a, int p)
{
    int res = 1;
    while(p)
    {
        if(p & 1)res = res * a % mod;
        a = a * a % mod;
        p >>= 1;
    }
    return res % mod;
}
void init()
{
    f[0] = g[0] = 1;
    for(int i = 1; i < N; i++)
    {
        f[i] = f[i - 1] * i % mod;
        g[i] = g[i - 1] * qpow(i, mod - 2) % mod;
    }
     
}
int C(int x, int y)
{
    return f[x] * g[y] % mod * g[x - y] % mod;
}
int A(int x, int y)
{
    if(y == 0) return 1;
    int ans = 1;
    for(int i = x; i > x - y; i --)
    {
        ans = ans * i % mod;
    }
    return ans % mod;
}
